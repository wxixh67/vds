# 允许所有搜索引擎访问
User-agent: *

# 允许索引首页和其它特定的 HTML 页面
Allow: /index.html
Allow: /note.html
Allow: /work.html
Allow: /money.html

# 2fa 文件夹中只有 index.php 可以被索引

Disallow: /app/

# assets 文件夹通常不需要被搜索引擎索引，但这不会影响搜索引擎读取这些资源以渲染页面
Disallow: /assets/

# 通常，404 页面不需要被索引
Disallow: /404.html
