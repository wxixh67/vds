import os

# 定义图片文件夹和目标文件路径
image_folder = r"C:\Users\l\Documents\GitHub\vds\p\images"
output_file = r"C:\Users\l\Documents\GitHub\vds\p\ttt.txt"

# 获取图片文件列表（可以根据需要过滤扩展名）
image_extensions = ['.png', '.jpg', '.jpeg']
image_files = [f for f in os.listdir(image_folder) if any(f.lower().endswith(ext) for ext in image_extensions)]

# 生成 img 标签列表
img_tags = []
for image_file in image_files:
    # 获取文件名（去除扩展名）
    file_name = os.path.splitext(image_file)[0]
    
    # 构造 img 标签
    img_tag = f'<img src="https://raw.githubusercontent.com/wxixh67/vds/main/p/images/{file_name}.png" alt="{file_name}" width="300">'
    img_tags.append(img_tag)

# 将 img 标签写入 ttt.txt 文件
with open(output_file, 'w') as file:
    file.write("\n".join(img_tags))

print(f"图片的 HTML 标签已写入到 {output_file}")
